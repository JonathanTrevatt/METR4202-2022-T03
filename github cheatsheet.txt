#Install git for windows
	https://gitforwindows.org/

#Open git gui
#Create ssh keys
#Add private key to your git account settings ssh keys

#Open terminal to your local working folder

#Set up your git details
	git config --global user.email "you@example.com"
	git config --global user.name "Your Name"
	
#Clone repository for the first time into current directory
	git clone https://github.com/JonathanTrevatt/METR4202.git

#Update your working directory with any committed changes
	git pull

#add all files in current directory to staging area
	git add .

#Commit files from staging area to the repository
	git commit -m "You commit message here"
